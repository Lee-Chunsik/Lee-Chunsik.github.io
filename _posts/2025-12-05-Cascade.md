해당 정보를 바탕으로 패스워드 복호화 `Python`스크립트를 작성했다.

```python
#!/usr/bin/env python3
import base64
from Crypto.Cipher import AES

KEY = b"c4scadek3y654321"      # 앞서 확인한 key 값
IV  = b"1tdyjCbY1Ix49842"      # 앞서 확인한 IV 값

def pkcs7_unpad(data: bytes) -> bytes:        # Padding(PKCS7) 제거
    pad_len = data[-1]
    if pad_len < 1 or pad_len > AES.block_size:        # 패딩 값이 이상하면 그대로 반환
        return data
    return data[:-pad_len]

def decrypt_pwd(enc_b64: str) -> str:
    ciphertext = base64.b64decode(enc_b64)        # Convert.FromBase64String(text4)

    cipher = AES.new(KEY, AES.MODE_CBC, iv=IV)
    padded_plain = cipher.decrypt(ciphertext)

    plain = pkcs7_unpad(padded_plain)
    return plain.decode("utf-8")

if __name__ == "__main__":
    enc = input("base64 암호 문자열(Pwd) 입력: ").strip()
    print("Decrypted:", decrypt_pwd(enc))
```

해당 스크립트는 `base64 암호 문자열`즉, `sqlite3`에서 확인한 `ArkSvc`의 패스워드를 입력하면 복호화하여 결과를 출력한다.
